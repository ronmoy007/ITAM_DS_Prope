  1. Encontrar los histogramas de probabilidad para las distribuciones asociadas a las siguientes variables aleatorias:
  1. El n√∫mero de √°guilas que aparecen cuando lanzamos tres monedas al aire.
a0<-dbinom(0,3,.5)
a1<-dbinom(1,3,.5)
a2<-dbinom(2,3,.5)
a3<-dbinom(3,3,.5)

df = data.frame(nbrAguilas=c('0 √°guilas', '1 √°guila', '2 √°guilas', '3 √°guilas'), X=c(0,1,2,3), probabilidad=c(a0,a1,a2,a3))

ggplot(data=df, aes(x=X, y=probabilidad, fill=nbrAguilas)) +
geom_col(width=1) +
ggtitle('Histograma')

  1. La suma de los n√∫meros de las caras cuando se lanzan dos dados.
S <- rolldie(2, nsides=6, makespace = TRUE)
S <- addrv(S, U=X1+X2)
sumas <- marginal(S,vars='U')

df = data.frame(sumaCaras = factor(sumas$U), probabilidad=sumas$probs)
ggplot(data=df, aes(x=sumaCaras, y=probabilidad, fill=sumaCaras)) + 
geom_col(width=1) +
ggtitle('Histograma')
